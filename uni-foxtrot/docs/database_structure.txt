=== INKY DATABASE STRUCTURE ===

TABLE: intents
----------------------------------------
  0   int_id               INTEGER NOT NULL PRIMARY KEY DEFAULT NULL
  1   int_name             TEXT NOT NULL DEFAULT NULL
  2   int_protocol         TEXT NOT NULL DEFAULT NULL

TABLE: messages
----------------------------------------
  0   msg_id               INTEGER NOT NULL PRIMARY KEY DEFAULT NULL
  1   sum_id               INTEGER NOT NULL DEFAULT 0
  2   msg_role             TEXT NOT NULL DEFAULT NULL
  3   msg_content          TEXT NOT NULL DEFAULT NULL
  4   msg_type             TEXT NOT NULL DEFAULT 'llm'
  5   msg_metadata         TEXT NOT NULL DEFAULT ''
  6   int_id               INTEGER NOT NULL DEFAULT 0
  7   msg_created          timestamp NOT NULL DEFAULT current_timestamp
  8   msg_updated          timestamp NOT NULL DEFAULT current_timestamp

TABLE: regex_patterns
----------------------------------------
  0   pat_id               INTEGER NOT NULL PRIMARY KEY DEFAULT NULL
  1   pat_regex            TEXT NOT NULL DEFAULT ''
  2   int_id               INTEGER NOT NULL DEFAULT 0
  3   pat_created          TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP
  4   pat_updated          TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP

TABLE: sqlean_define
----------------------------------------
  0   name                 TEXT PRIMARY KEY DEFAULT NULL
  1   type                 TEXT DEFAULT NULL
  2   body                 TEXT DEFAULT NULL

TABLE: summaries
----------------------------------------
  0   sum_id               INTEGER NOT NULL PRIMARY KEY DEFAULT NULL
  1   sum_title            TEXT NOT NULL DEFAULT ''
  2   sum_content          TEXT NOT NULL DEFAULT ''
  3   sum_level            INTEGER NOT NULL DEFAULT 0
  4   sum_created          TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP
  5   sum_updated          TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP

TABLE: user_entities
----------------------------------------
  0   ent_id               INTEGER NOT NULL PRIMARY KEY DEFAULT NULL
  1   user_id              INTEGER NOT NULL DEFAULT NULL
  2   ent_name             TEXT NOT NULL DEFAULT ''
  3   ent_given_name       TEXT NOT NULL DEFAULT ''
  4   ent_categories       TEXT NOT NULL DEFAULT ''
  5   ent_created          TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP
  6   ent_updated          TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP
